# ==================== DLA test_chunking_gradio_diarization.py ====================

# Wybór silnika diarization
# true = Deepgram API (zalecane, szybkie, wymaga klucza API)
# false = whisper-diarization lokalne (wolniejsze, wymaga CUDA GPU)
USE_DEEPGRAM=true

# --- Konfiguracja Deepgram API (jeśli USE_DEEPGRAM=true) ---
# Klucz API Deepgram - WYMAGANE jeśli USE_DEEPGRAM=true
# Pobierz z: https://console.deepgram.com/
DEEPGRAM_API_KEY=your_deepgram_api_key_here

# --- Konfiguracja whisper-diarization (jeśli USE_DEEPGRAM=false) ---
# Klucz API Groq - WYMAGANE jeśli USE_DEEPGRAM=false
# Pobierz z: https://console.groq.com/keys
GROQ_API_KEY=your_groq_api_key_here

# Ścieżka do katalogu whisper-diarization (tylko jeśli USE_DEEPGRAM=false)
# Domyślnie: ./whisper-diarization
# Przykłady:
# WHISPER_DIARIZATION_DIR=./whisper-diarization
# WHISPER_DIARIZATION_DIR=C:\Users\MSI\PycharmProjects\whisper-diarization
WHISPER_DIARIZATION_DIR=./whisper-diarization

# Urządzenie dla whisper-diarization (tylko jeśli USE_DEEPGRAM=false)
# cuda = użyj GPU (szybsze, wymaga CUDA)
# cpu = użyj CPU (wolniejsze, ale zawsze działa)
DEVICE=cuda

# Model Whisper dla whisper-diarization (tylko jeśli USE_DEEPGRAM=false)
# Opcje: base, small, medium, large-v3
# Dla GTX 1050 Ti (4GB VRAM) zalecane: small lub medium
# Dla większych GPU: medium lub large-v3
WHISPER_MODEL=small

# Wymuszenie CUDA (pomiń sprawdzanie dostępności)
# true = wymuś CUDA bez sprawdzania (może nie działać)
# false = sprawdź dostępność CUDA przed użyciem (zalecane)
FORCE_CUDA=false

# Język transkrypcji
# Opcje: pl (polski), en (angielski), auto (automatyczne wykrywanie)
# Domyślnie: pl
LANG=pl

# ==================== DLA test_chunking_gradio.py ====================

# Klucz API Groq - WYMAGANE dla test_chunking_gradio.py
# Pobierz z: https://console.groq.com/keys
# GROQ_API_KEY=your_groq_api_key_here

# Klucz API OpenRouter - WYMAGANE dla test_chunking_gradio.py
# Pobierz z: https://openrouter.ai/keys
OPENROUTER_API_KEY=your_openrouter_api_key_here

# Model Gemini do użycia (opcjonalne)
# Domyślnie: google/gemini-2.5-flash-lite
# Inne opcje:
#   - google/gemini-2.0-flash-exp (najnowszy, eksperymentalny)
#   - google/gemini-1.5-flash (stabilny)
#   - google/gemini-2.5-flash-lite (najszybszy, zalecany)
GEMINI_FLASH_MODEL=google/gemini-2.5-flash-lite

# Język transkrypcji (opcjonalne)
# Domyślnie: pl
# LANG=pl
